#include <iostream>
#include <iomanip> // For formatting decimal numbers
using namespace std;

int main() {
    // Variables to store input
    string itemName;
    int quantity;
    float cost;

    // Get item name (can include spaces)
    cout << "Enter item name: ";
    cin.ignore(); // Clears the newline character from previous input
    getline(cin, itemName);

    // Get quantity and cost
    cout << "Enter quantity: ";
    cin >> quantity;

    cout << "Enter cost per item: $";
    cin >> cost;

    // Calculate total cost
    float totalCost = quantity * cost;

    // Display total cost with 2 decimal places
    cout << fixed << setprecision(2);
    cout << "Total cost for \"" << itemName << "\": $" << totalCost << endl;

    return 0;
}
